void main() {
  DateTime data = DateTime.now();
  DateTime entrada1 = DateTime(data.year, data.month, data.day, 08, 22);
  DateTime saida1 = DateTime(data.year, data.month, data.day, 13, 00);
  DateTime entrada2 = DateTime(data.year, data.month, data.day, 14, 56);
  DateTime saida2 = DateTime(data.year, data.month, data.day, 18, 38);

  int resultMinutos1 = saida1.difference(entrada1).inMinutes;

  int resultMinutos2 = saida2.difference(entrada2).inMinutes;

  int horasTrabalhadasMinutos = resultMinutos1 + resultMinutos2;
    int horasExtras = horasTrabalhadasMinutos - 480;
  print(durationToString(horasTrabalhadasMinutos));
  
  print(horasExtra(horasExtras));
}

String durationToString(int minutes) {
  var d = Duration(minutes: minutes);
  List<String> parts = d.toString().split(':');
  return '${parts[0].padLeft(2, '0')}:${parts[1].padLeft(2, '0')}';
}

String horasExtra(int minutes) {
  var d = Duration(minutes: minutes);
  List<String> parts = d.toString().split(':');
  return '${parts[0].padLeft(2, '0')}:${parts[1].padLeft(2, '0')}';
}
